Nested GridSearch to Find Overall Best performing Model¶

In [291]:
non_nested_scores = np.zeros(5)
nested_scores = np.zeros(5)

# Setting up Classifiers and parameter dictionaries
svm_params = [{'kernel': ['rbf'], 'gamma': [1e-3, 1e-4],'C': np.arange(0.01,3,0.5)},
              {'kernel': ['linear'], 'C': np.arange(0.01,3,0.5)}]
svm = SVC()

dt = DecisionTreeClassifier(random_state = 10)
dt_grid = {'max_depth':list(range(1,15)), 
           'criterion':['gini', 'entropy'],
           'min_samples_split':list(range(2,15)), 
           'min_samples_leaf':list(range(2,15))}

knn = KNeighborsClassifier()
knn_grid = {'n_neighbors':list(range(1,20)), 
            'weights':['uniform', 'distance'], 
            'algorithm':['auto', 'ball_tree', 'kd_tree', 'brute']}

logit = LogisticRegression(random_state=10, multi_class= 'multinomial', solver='saga')
lt_grid = {'penalty':['l1','l2'], 'C':list(np.arange(1, 10, 1))}

nb = MultinomialNB()
nb_grid = [{'alpha': np.arange(1,10,0.5)}]

classifiers = [logit, knn, dt, svm, nb]
dicts = [lt_grid, knn_grid, dt_grid, svm_params, nb_grid]

# Nested cross-validation
for i in range(5):
    inner_cv = KFold(n_splits=4, shuffle=True, random_state=42)
    outer_cv = KFold(n_splits=4, shuffle=True, random_state=42)

    # Non_nested parameter search and scoring
    clf = GridSearchCV(estimator=classifiers[i], param_grid=dicts[i], cv=inner_cv,iid=False, n_jobs = -1)
    clf.fit(x_train, y_train)
    non_nested_scores[i] = clf.best_score_

    # Nested CV with parameter optimization
    nested_score = cross_val_score(clf, X=x_train, y=y_train, cv=outer_cv)
    nested_scores[i] = nested_score.mean()
    
print(nested_scores)

    # Non_nested parameter search and scoring

    # Nested CV with parameter optimization
    nested_score = cross_val_score(clf, X=x_train, y=y_train, cv=outer_cv)
    nested_score[i] = nested_score.mean()
